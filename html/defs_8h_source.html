161#ifndef clamp
162#if !defined(__cplusplus)
163#define clamp(x, y, z) \
164 __extension__ ({ \
165 __typeof__ (x) _x = (x); \
166 __typeof__ (y) _y = (y); \
167 __typeof__ (z) _z = (z); \
168 (_x <= _y) ? _y : ((_x >= _z) ? _z : _x); \
169 })
170#else
171template<typename _T>
172constexpr const _T& clamp(const _T& _value, const _T& _min, const _T& _max) {
173 return (_value < _min) ? _min : (_max < _value) ? _max : _value;
174}
175
176template<typename _T, typename _Comparer>
177constexpr const _T&
178clamp(const _T& _value, const _T& _min, const _T& _max, _Comparer _compare) {
179 return _compare(_value, _min) ? _min : _compare(_max, _value) ? _max : _value;
180}
181#endif
182#endif /* !clamp */
