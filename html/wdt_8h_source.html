102static inline void __always_inline
103cpu_wdt_enable(void)
104{
105 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
106
107 uint8_t wtcr_bits;
108 wtcr_bits = cpu_ioregs->wtcsrr;
109
110 wtcr_bits &= ~0x80;
111 wtcr_bits |= 0x20;
112
113 cpu_ioregs->wtcsrw = CPU_WDT_WTCSR(wtcr_bits);
114}
