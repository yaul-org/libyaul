69static inline void __always_inline
70smpc_smc_wait(bool enter)
71{
72 volatile uint8_t * const reg_sf = (volatile uint8_t *)SMPC(SF);
73
74 while ((*reg_sf & 0x01) == 0x01) {
75 }
76
77 if (enter) {
78 *reg_sf = 0x01;
79 }
80}

90static inline uint8_t __always_inline
91smpc_smc_call(cpu_smpc_cmd_t cmd)
92{
93 smpc_smc_wait(true);
94
95 MEMORY_WRITE(8, SMPC(COMREG), cmd);
96
97 smpc_smc_wait(false);
98
99 return MEMORY_READ(8, OREG(31));
100}

158static inline uint8_t __always_inline
159smpc_smc_intback_call(uint8_t ireg0, uint8_t ireg1)
160{
161 /* Fetch peripheral data "SMPC" status */
162 smpc_smc_wait(true);
163
164 MEMORY_WRITE(8, IREG(0), ireg0 & 0x01);
165 MEMORY_WRITE(8, IREG(1), ireg1 & 0xFA);
166 MEMORY_WRITE(8, IREG(2), 0xF0);
167
168 MEMORY_WRITE(8, SMPC(COMREG), SMPC_SMC_INTBACK);
169
170 return 0;
171}
