69/*-
70 * Possible values for VDP2_SCRN_PAGE_SIZE_CALCULATE() (in bytes):
71 * +----------+-----------+---------------+
72 * | PND size | Cell size | Size of page |
73 * +----------+-----------+---------------+
74 * | 1-word | 2x2 | 0x0800 |
75 * | 2-word | 2x2 | 0x1000 |
76 * | 1-word | 1x1 | 0x2000 |
77 * | 2-word | 1x1 | 0x4000 |
78 * +----------+-----------+---------------+
79 *
80 * Page dimension is 64x64 if cell size is 1x1.
81 * Page dimension is 32x32 if cell size is 2x2 */
85#define VDP2_SCRN_PAGE_SIZE_CALCULATE(format) \
86 (VDP2_SCRN_PAGE_DIMENSION_CALCULATE(format) * ((format)->pnd_size * 2))
