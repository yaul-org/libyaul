75/* The __CONCAT macro is used to concatenate parts of symbol names, e.g. with
76 * "#define OLD(foo) __CONCAT(old,foo)", OLD(foo) produces oldfoo. The __CONCAT
77 * macro is a bit tricky to use if it must work in non-ANSI mode -- there must
78 * be no spaces between its arguments, and for nested __CONCAT's, all the
79 * __CONCAT's must be at the left. __CONCAT can also concatenate double-quoted
80 * strings produced by the __STRING macro, but this only works with ANSI C.
81 *
82 * __XSTRING is like __STRING, but it expands any macros in its argument first.
83 * It is only available with ANSI C. */
84
85#define __CONCAT1(x,y) x ## y
86#define __CONCAT(x,y) __CONCAT1(x,y)
87#define __STRING(x) #x /* Stringify without expanding x */
88#define __STRINGIFY(x) #x /* Stringify without expanding x */
89#define __XSTRING(x) __STRING(x) /* Expand x, then stringify */

250/* C99 Static array indices in function parameter declarations. Syntax such as:
251 *
252 * void bar(int myArray[static 10]);

254 * is allowed in C99 but not in C++.

256 * Define __min_size appropriately so
257 * headers using it can be compiled in either language. Use like this:
258 *
259 * void bar(int myArray[__min_size(10)]); */

265#endif /* !__min_size */

267#ifndef __min_size
268#define __min_size(x) (x)

272/* The identifier __func__ is implicitly declared by the translator as if,
273 * immediately following the opening brace of each function definition, the
274 * declaration:
275 *
276 * static const char __func__[] = "function-name";
277 *
278 * appeared, where function-name is the name of the lexically-enclosing
279 * function. This name is the unadorned name of the function. As an extension,
280 * at file (or, in C++, namespace scope), __func__ evaluates to the empty
281 * string. */
282#ifndef __func__
283#define __func__ ((const char *)0)
284#endif /* !__func__ */
285
286#ifndef __function_name
287/* Use g++'s demangled names in C++ */
288#if defined(__cplusplus) && defined(__GNUC__)
289#define __function_name __PRETTY_FUNCTION__
290/* C99 requires the use of __func__ */
291#elif __STDC_VERSION__ >= 199901L
292#define __function_name __func__
293/* Failed to detect __func__ support */
294#else
295#define __function_name ((const char *)0)
296#endif
297#endif /* !__function_name */

313/* GNU C version 2.96 adds explicit branch prediction so that the CPU back-end
314 * can hint the processor and also so that code blocks can be reordered such
315 * that the predicted path sees a more linear flow, thus improving cache
316 * behavior, etc.
317 *
318 * The following two macros provide us with a way to utilize this compiler
319 * feature. Use __predict_true() if you expect the expression to evaluate to
320 * true, and __predict_false() if you expect the expression to evaluate to
321 * false.
322 *
323 * A few notes about usage:
324 *
325 * * Generally, __predict_false() error condition checks (unless you have some
326 * _strong_ reason to do otherwise, in which case document it), and/or
327 * __predict_true() `no-error' condition checks, assuming you want to
328 * optimize for the no-error case.
329 *
330 * * Other than that, if you don't know the likelihood of a test succeeding
331 * from empirical or other `hard' evidence, don't make predictions.
332 *
333 * * These are meant to be used in places that are run `a lot'. It is wasteful
334 * to make predictions in code that is run seldomly (e.g. at subsystem
335 * initialization time) as the basic block reordering that this affects can
336 * often generate larger code. */
337#ifndef __predict_true
338#define __predict_true(exp) __builtin_expect((exp), 1)
339#endif /* __predict_true */
340#ifndef __predict_false
341#define __predict_false(exp) __builtin_expect((exp), 0)
342#endif /* !__predict_false */
