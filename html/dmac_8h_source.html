316static inline void __always_inline

336static inline uint8_t __always_inline
337cpu_dmac_interrupt_priority_get(void)
338{
339 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
340
341 return ((cpu_ioregs->ipra >> 8) & 0x0F);
342}

347static inline void __always_inline
348cpu_dmac_interrupt_priority_set(uint8_t priority)
349{
350 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
351
352 cpu_ioregs->ipra &= 0xF0FF;
353 cpu_ioregs->ipra |= (priority & 0x0F) << 8;
354}

359static inline void __always_inline
360cpu_dmac_priority_mode_set(cpu_dmac_priority_mode_t mode)
361{
362 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
363
364 cpu_ioregs->dmaor &= ~0x00000001;
365 cpu_ioregs->dmaor |= (mode & 0x01) << 3;
366}

376static inline void __always_inline
377cpu_dmac_channel_start(cpu_dmac_channel_t ch)
378{
379 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;

381 cpu_ioregs->channels[ch].chcrn &= ~0x00000003;

388static inline void __always_inline

390{
391 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
392
393 /* Don't clear the status bits */
394 cpu_ioregs->channels[ch].chcrn &= ~0x00000001;
395}
