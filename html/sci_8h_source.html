269static inline uint8_t __always_inline
270cpu_sci_read_value_get(void)
271{
272 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
273
274 const uint8_t value = cpu_ioregs->rdr;
275
276 /* If RDRF is set, clear it */
277 if ((cpu_ioregs->ssr & 0x40) > 0) {
278 cpu_ioregs->ssr &= ~0x40;
279 }
280
281 return value;
282}
