184static inline uint8_t __always_inline
185cpu_intc_mask_get(void)
186{
187 __register const uint32_t reg_sr = cpu_reg_sr_get();
188 __register const uint32_t mask = (reg_sr & CPU_SR_I_BITS_MASK) >> 4;
189
190 return mask;
191}

196static inline void __always_inline
197cpu_intc_mask_set(uint8_t mask)
198{
199 __register uint32_t reg_sr;
200 reg_sr = cpu_reg_sr_get();
201
202 reg_sr &= ~CPU_SR_I_BITS_MASK;
203 reg_sr |= (mask << 4) & CPU_SR_I_BITS_MASK;
204
205 cpu_reg_sr_set(reg_sr);
206}
