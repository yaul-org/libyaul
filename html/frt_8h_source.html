91static inline uint16_t __always_inline
92cpu_frt_count_get(void)
93{
94 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
95
96 const uint16_t reg_frth = cpu_ioregs->frch;
97 const uint16_t reg_frtl = cpu_ioregs->frcl;

99 return (reg_frth << 8) | reg_frtl;

132static inline void __always_inline
133cpu_frt_interrupt_priority_set(uint8_t priority)
134{
135 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
136
137 /* Set the interrupt priority level for FRT (shared amongst all FRT
138 * related interrupts */
139 cpu_ioregs->iprb &= 0xF0FF;
140 cpu_ioregs->iprb |= (priority & 0x0F) << 8;
141}
