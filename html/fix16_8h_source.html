38 /* Macro for defining fix16_t constant values. The functions above can't be used
39  * from e.g. global variable initializers, and their names are quite long also.
40  * This macro is useful for constants springled alongside code, e.g.
41  * FIX16(1.234).
42  *
43  * Note that the argument is evaluated multiple times, and also otherwise you
44  * should only use this for constant values. For runtime-conversions, use the
45  * inlined functions below */
46 
47 #define FIX16(x) ((fix16_t)(((x) >= 0) \
48  ? ((x) * 65536.0f + 0.5f) \
49  : ((x) * 65536.0f - 0.5f)))

101 static inline fix16_t __always_inline
102 fix16_mul(const fix16_t a, const fix16_t b)
103 {
104  register uint32_t mach;
105  register fix16_t out;
106 
107  __asm__ volatile ("\tdmuls.l %[a], %[b]\n"
108  "\tsts mach, %[mach]\n"
109  "\tsts macl, %[out]\n"
110  "\txtrct %[mach], %[out]\n"
111  /* Output */
112  : [mach] "=&r" (mach),
113  [out] "=&r" (out)
114  /* Input */
115  : [a] "r" (a),
116  [b] "r" (b)
117  : "mach", "macl");
118 
119  return out;
120 }
