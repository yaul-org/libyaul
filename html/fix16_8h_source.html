149static inline int32_t __always_inline
150fix16_low_mul(fix16_t a, fix16_t b)
151{
152 __register int32_t out;
153
154 __declare_asm("\tdmuls.l %[a], %[b]\n"
155 "\tsts macl, %[out]\n"
156 /* Output */
157 : [out] "=&r" (out)
158 /* Input */
159 : [a] "r" (a),
160 [b] "r" (b)
161 : "macl");
162
163 return out;
164}

198__BEGIN_ASM
199static inline fix16_t __always_inline
200fix16_mul(fix16_t a, fix16_t b)
201{
202 __register uint32_t mach;
203 __register fix16_t out;
204
205 __declare_asm("\tdmuls.l %[a], %[b]\n"
206 "\tsts mach, %[mach]\n"
207 "\tsts macl, %[out]\n"
208 "\txtrct %[mach], %[out]\n"
209 /* Output */
210 : [mach] "=&r" (mach),
211 [out] "=&r" (out)
212 /* Input */
213 : [a] "r" (a),
214 [b] "r" (b)
215 : "mach", "macl");
216
217 return out;
218}
219__END_ASM

249static inline int32_t __always_inline
250fix16_round_int32_to(fix16_t value)
251{
252 if (value >= 0) {
253 return ((value + (FIX16(1.0) / 2)) / FIX16(1.0));
254 }
255
256 return ((value - (FIX16(1.0) / 2)) / FIX16(1.0));
257}

468inline int16_t fix16_t::round() const {
469 // Avoid use of fix16_t operations like division as that would use the CPU-DIVU
470 if (value >= 0) {
471 return ((value + 32768) / 65536);
472 }
473
474 return ((value - 32768) / 65536);
