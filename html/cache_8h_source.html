144static inline void __always_inline
145cpu_cache_repl_enable(cpu_cache_type_t type)
146{
147 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
148
149 const uint8_t t0 = cpu_ioregs->ccr & ~((uint8_t)type | 0x01);
150
151 cpu_ioregs->ccr = t0;
152 cpu_ioregs->ccr = t0 | 0x01;
153}

161static inline void __always_inline
162cpu_cache_repl_disable(cpu_cache_type_t type)
163{
164 volatile cpu_ioregs_t * const cpu_ioregs = (volatile cpu_ioregs_t *)CPU_IOREG_BASE;
165
166 const uint8_t t0 = cpu_ioregs->ccr & ~0x01;
167
168 /* Set bit with cache disabled */
169 cpu_ioregs->ccr = t0 | (uint8_t)type;
170 /* Enable cache and set bit(s) */
171 cpu_ioregs->ccr = t0 | (uint8_t)type | 0x01;
172}
