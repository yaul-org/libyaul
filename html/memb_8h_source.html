50#define MEMB(name, structure, count, align) \
51static struct memb_ref \
52 __CONCAT(name, _memb_ref)[((count) <= 0) ? 1 : (count)]; \
53 \
54static __aligned(((align) <= 0) ? 4 : (align)) \
55 structure __CONCAT1(name, _memb_mem)[((count) <= 0) ? 1 : (count)]; \
56 \
57static memb_t name = { \
58 MEMB_TYPE_STATIC, \
59 sizeof(structure), \
60 ((count) <= 0) ? 1 : (count), \
61 &__CONCAT(name, _memb_ref)[0], \
62 0, \
63 0, \
64 (void *)&__CONCAT(name, _memb_mem)[0], \
65 NULL \
66}
