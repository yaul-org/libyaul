193static inline void __always_inline
194vdp2_tvmd_vcount_wait(uint16_t scanline)
195{
196 /* There are 1024 maximum counter values */
197 scanline &= 0x03FF;
198
199 while (true) {
200 vdp2_tvmd_extern_latch();
201
202 if ((vdp2_tvmd_vcount_get()) == scanline) {
203 break;
204 }
205 }
