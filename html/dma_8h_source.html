294static inline uint32_t __always_inline
295scu_dma_dsp_busy(void)
296{
297 volatile scu_ioregs_t * const scu_ioregs =
298 (volatile scu_ioregs_t *)SCU_IOREG_BASE;

376static inline uint32_t __always_inline
377scu_dma_level_busy(scu_dma_level_t level)
378{
379 switch (level & SCU_DMA_LEVEL_COUNT) {
380 case 1:
381 return scu_dma_level1_busy();
382 case 2:
383 return scu_dma_level2_busy();
384 case 0:
385 default:
386 return scu_dma_level0_busy();
387 }
388}
