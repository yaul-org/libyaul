94cpu_instr_macw(void *a, void *b)
95{
96 __register uint16_t **ap = (uint16_t **)a;
97 __register uint16_t **bp = (uint16_t **)b;
98
99 __declare_asm("mac.w @%[a]+, @%[b]+"
100 : [a] "+&r" (*ap),
