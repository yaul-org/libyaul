43static inline bool __always_inline
44cpu_sync_mutex(cpu_sync_lock_t b)
45{
46 __register uint8_t *bios_address;
47 bios_address = (uint8_t *)HWRAM_UNCACHED(0x00000B00);
48
49 __register uint32_t result;
50 result = 0;
51
52 /* Currently using cpu_sync_mutex() to compare the value results in
53 * extra instructions being emitted. Because TAS.B sets the T bit in SR,
54 * there shouldn't be a need for the MOVT instruction */
55
56 __declare_asm("\tadd %[b], %[bios_address]\n"
57 "\ttas.b @%[out]\n"
58 "\tmovt %[result]"
59 /* Output */
60 : [out] "=r" (bios_address),
61 [result] "=r" (result)
62 /* Input */
63 : [bios_address] "r" (bios_address),
64 [b] "r" (b));
65
66 return result;
67}
