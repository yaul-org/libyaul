241fix16_vec3_inline_dot(const fix16_vec3_t *a, const fix16_vec3_t *b)
242{
243 __register uint32_t aux0;
244 __register uint32_t aux1;
245
246 __declare_asm("\tclrmac\n"
247 "\tmac.l @%[a]+, @%[b]+\n"
248 "\tmac.l @%[a]+, @%[b]+\n"
249 "\tmac.l @%[a]+, @%[b]+\n"
250 "\tsts mach, %[aux0]\n"
251 "\tsts macl, %[aux1]\n"
252 "\txtrct %[aux0], %[aux1]\n"
253 : [a] "+r" (a),
254 [b] "+r" (b),
255 [aux0] "=&r" (aux0),
256 [aux1] "=&r" (aux1)
257 : "m" (*a),
258 "m" (*b)
259 : "mach", "macl", "memory");
260
261 return aux1;
262}
