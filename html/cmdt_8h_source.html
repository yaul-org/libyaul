372static inline void __always_inline
373vdp1_cmdt_draw_mode_set(vdp1_cmdt_t *cmdt,
374 vdp1_cmdt_draw_mode_t draw_mode)
375{
376 /* Values 0x4, 0x5, 0x6 for comm indicate a non-textured command table,
377 * and we want to set the bits 7 and 6 without branching */
378 const uint16_t comm = (cmdt->cmd_ctrl & 0x0004);
379 const uint16_t pmod_bits = (comm << 5) | (comm << 4);
380
381 cmdt->cmd_draw_mode.raw = pmod_bits | draw_mode.raw;
382}
