44static inline uint16_t __always_inline
45cpu_bswap16(uint16_t x)
46{
47 __register uint32_t out;
48
49 __asm__ volatile ("swap.b %[in], %[out]\n"
50 : [out] "=&r" (out)
51 : [in] "r" (x));
52

54}

75 __declare_asm("swap.b %[in], %[ot0]\n"

81 : [in] "0" "r" (x),
82 [it0] "1" (t0),
